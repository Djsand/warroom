# gaps MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a CLI tool where 5 AI agents (Architect, Challenger, Builder, Reviewer, Breaker) debate, build, and review code — producing a readable conversation.md as the core output.

**Architecture:** A TypeScript CLI (`gaps run "task"`) that orchestrates sequential Claude API calls, each agent with a role-specific system prompt and different objective function. Agents communicate through a shared conversation thread. The Builder agent outputs file contents in a structured format. Output: conversation.md (the viral artifact), summary.md (morning report), and code changes on a git branch.

**Tech Stack:** TypeScript, Node.js, @anthropic-ai/sdk, Commander.js, simple-git, chalk, Vitest

```typescript
// src/conversation/thread.ts
import type { AgentRole, Phase, ConversationMessage, CodeChange, ConversationStats } from "../types.js";
import { AGENT_LABEL, AGENT_EMOJI } from "../types.js";

interface AddMessageInput {
  role: AgentRole;
  content: string;
  phase: Phase;
  codeChanges?: CodeChange[];
}

export class Thread {
  readonly task: string;
  readonly messages: ConversationMessage[] = [];
  readonly startedAt: Date;

  constructor(task: string) {
    this.task = task;
    this.startedAt = new Date();
  }

  add(input: AddMessageInput): ConversationMessage {
    const message: ConversationMessage = {
      role: input.role,
      content: input.content,
      phase: input.phase,
      timestamp: new Date(),
      codeChanges: input.codeChanges,
    };
    this.messages.push(message);
    return message;
  }

  messagesForPhase(phase: Phase): ConversationMessage[] {
    return this.messages.filter((m) => m.phase === phase);
  }

  toPromptContext(): string {
    return this.messages
      .map((m) => `**${AGENT_EMOJI[m.role]} ${AGENT_LABEL[m.role]}:** ${m.content}`)
      .join("\n\n");
  }

  getStats(): ConversationStats {
    const allCodeChanges = this.messages.flatMap((m) => m.codeChanges ?? []);
    const totalLines = allCodeChanges.reduce(
      (sum, c) => sum + c.content.split("\n").length,
      0
    );

    return {
      totalMessages: this.messages.length,
      designRevisions: this.messagesForPhase("design")
        .filter((m) => m.role === "architect").length,
      bugsCaught: this.messagesForPhase("review").filter(
        (m) => m.content.toLowerCase().includes("bug") ||
               m.content.toLowerCase().includes("issue") ||
               m.content.toLowerCase().includes("found")
      ).length,
      filesChanged: allCodeChanges.length,
      linesAdded: totalLines,
      durationMs: Date.now() - this.startedAt.getTime(),
    };
  }
}
```

```typescript
// src/conversation/renderer.ts
import type { Thread } from "./thread.js";
import type { Phase } from "../types.js";
import { AGENT_EMOJI, AGENT_LABEL } from "../types.js";

const PHASE_LABELS: Record<Phase, string> = {
  design: "Design",
  build: "Build",
  review: "Review",
  finalize: "Finalize",
};

const PHASE_ORDER: Phase[] = ["design", "build", "review", "finalize"];

export function renderConversation(thread: Thread): string {
  const lines: string[] = [];

  lines.push(`# gaps conversation \u2014 "${thread.task}"`);
  lines.push("");

  const participants = [...new Set(thread.messages.map((m) => m.role))];
  if (participants.length > 0) {
    lines.push("## Participants");
    for (const role of participants) {
      lines.push(`- ${AGENT_EMOJI[role]} **${AGENT_LABEL[role]}**`);
    }
    lines.push("");
  }

  let phaseNum = 0;
  for (const phase of PHASE_ORDER) {
    const phaseMessages = thread.messagesForPhase(phase);
    if (phaseMessages.length === 0) continue;

    phaseNum++;
    const durationMs =
      phaseMessages.length > 1
        ? phaseMessages[phaseMessages.length - 1].timestamp.getTime() -
          phaseMessages[0].timestamp.getTime()
        : 0;
    const durationMin = Math.max(1, Math.round(durationMs / 60000));

    lines.push(
      `## Phase ${phaseNum}: ${PHASE_LABELS[phase]} (${phaseMessages.length} messages, ~${durationMin} min)`
    );
    lines.push("");

    for (const msg of phaseMessages) {
      lines.push(
        `**${AGENT_EMOJI[msg.role]} ${AGENT_LABEL[msg.role]}:** ${msg.content}`
      );
      lines.push("");
    }
  }

  const stats = thread.getStats();
  const uniqueRoles = new Set(thread.messages.map((m) => m.role)).size;
  const durationMin = Math.max(1, Math.round(stats.durationMs / 60000));

  lines.push("---");
  lines.push("");
  lines.push(
    `${uniqueRoles} agents \u00B7 ${stats.totalMessages} messages \u00B7 ` +
    `${stats.designRevisions} design revisions \u00B7 ${stats.bugsCaught} bugs caught \u00B7 ` +
    `${stats.filesChanged} files \u00B7 ${stats.linesAdded} lines \u00B7 ~${durationMin} min`
  );

  return lines.join("\n");
}
```

```typescript
// src/conversation/summary.ts
import type { Thread } from "./thread.js";

export function renderSummary(thread: Thread, branchName: string): string {
  const stats = thread.getStats();
  const durationMin = Math.max(1, Math.round(stats.durationMs / 60000));
  const uniqueRoles = new Set(thread.messages.map((m) => m.role)).size;
  const lines: string[] = [];

  lines.push(`# gaps summary \u2014 ${thread.task}`);
  lines.push("");
  lines.push("## Result: COMPLETE");
  lines.push("");

  const designMessages = thread.messagesForPhase("design");
  if (designMessages.length > 0) {
    lines.push("### What the agents debated");
    for (const msg of designMessages) {
      if (msg.role === "challenger" || msg.role === "reviewer") {
        const preview = msg.content.slice(0, 120).replace(/\n/g, " ");
        lines.push(`- ${preview}${msg.content.length > 120 ? "..." : ""}`);
      }
    }
    lines.push("");
  }

  lines.push("### Stats");
  lines.push(`- ${uniqueRoles} agents, ${stats.totalMessages} messages, ~${durationMin} min`);
  lines.push(`- ${stats.designRevisions} architectural revisions`);
  lines.push(`- ${stats.bugsCaught} bugs/issues caught during review`);
  lines.push(`- ${stats.filesChanged} files changed, ${stats.linesAdded} lines added`);
  lines.push("");

  lines.push("### Branch");
  lines.push(`\`${branchName}\` \u2014 ready for your review`);

  return lines.join("\n");
}
```

```typescript
// src/agents/call.ts
import Anthropic from "@anthropic-ai/sdk";
import type { AgentRole, CodeChange } from "../types.js";

export interface CallAgentInput {
  role: AgentRole;
  systemPrompt: string;
  conversationContext: string;
  apiKey: string;
  model: string;
}

export interface CallAgentResult {
  content: string;
  codeChanges: CodeChange[];
  tokensUsed: number;
}

export async function callAgent(input: CallAgentInput): Promise<CallAgentResult> {
  const client = new Anthropic({ apiKey: input.apiKey });

  const response = await client.messages.create({
    model: input.model,
    max_tokens: 4096,
    system: input.systemPrompt,
    messages: [{ role: "user", content: input.conversationContext }],
  });

  const textContent = response.content
    .filter((block): block is Anthropic.TextBlock => block.type === "text")
    .map((block) => block.text)
    .join("\n");

  const codeChanges = extractCodeChanges(textContent);
  const tokensUsed =
    (response.usage?.input_tokens ?? 0) + (response.usage?.output_tokens ?? 0);

  return { content: textContent, codeChanges, tokensUsed };
}

function extractCodeChanges(text: string): CodeChange[] {
  const changes: CodeChange[] = [];
  const fileBlockRegex = /FILE:\s*([\w/.\\-]+)\s*\n```[\w]*\n([\s\S]*?)```/g;
  let match;
  while ((match = fileBlockRegex.exec(text)) !== null) {
    changes.push({
      filePath: match[1].trim(),
      content: match[2].trim(),
      action: "create",
    });
  }
  return changes;
}
```

```typescript
export function writeCodeChanges(baseDir: string, changes: CodeChange[]): void {
  for (const change of changes) {
    const fullPath = join(baseDir, change.filePath);
```

```typescript
// src/orchestrator/phases.ts
import { Thread } from "../conversation/thread.js";
import { callAgent } from "../agents/call.js";
import { getSystemPrompt } from "../agents/prompts.js";
import type { CodeChange } from "../types.js";

interface PhaseConfig {
  apiKey: string;
  architectModel: string;
  agentModel: string;
  maxRounds: number;
  projectContext: string;
}

export async function runDesignPhase(
  thread: Thread,
  config: PhaseConfig
): Promise<void> {
  const architectResult = await callAgent({
    role: "architect",
    systemPrompt: getSystemPrompt("architect", thread.task, config.projectContext),
    conversationContext: `Task: ${thread.task}\n\nProvide your architectural proposal.`,
    apiKey: config.apiKey,
    model: config.architectModel,
  });
  thread.add({ role: "architect", content: architectResult.content, phase: "design" });

  for (let round = 0; round < config.maxRounds; round++) {
    const challengerResult = await callAgent({
      role: "challenger",
      systemPrompt: getSystemPrompt("challenger", thread.task, config.projectContext),
      conversationContext: thread.toPromptContext(),
      apiKey: config.apiKey,
      model: config.agentModel,
    });
    thread.add({ role: "challenger", content: challengerResult.content, phase: "design" });

    if (challengerResult.content.toLowerCase().includes("design approved")) break;

    const revisionResult = await callAgent({
      role: "architect",
      systemPrompt: getSystemPrompt("architect", thread.task, config.projectContext),
      conversationContext: thread.toPromptContext(),
      apiKey: config.apiKey,
      model: config.architectModel,
    });
    thread.add({ role: "architect", content: revisionResult.content, phase: "design" });

    if (revisionResult.content.toLowerCase().includes("@builder")) break;
  }
}

export async function runBuildPhase(
  thread: Thread,
  config: PhaseConfig
): Promise<CodeChange[]> {
  const builderResult = await callAgent({
    role: "builder",
    systemPrompt: getSystemPrompt("builder", thread.task, config.projectContext),
    conversationContext:
      thread.toPromptContext() +
      "\n\nImplement the agreed design. Output each file using:\nFILE: path/to/file.ext\n```\ncode\n```",
    apiKey: config.apiKey,
    model: config.agentModel,
  });
  thread.add({
    role: "builder",
    content: builderResult.content,
    phase: "build",
    codeChanges: builderResult.codeChanges,
  });

  return builderResult.codeChanges;
}

export async function runReviewPhase(
  thread: Thread,
  config: PhaseConfig
): Promise<boolean> {
  const reviewerResult = await callAgent({
    role: "reviewer",
    systemPrompt: getSystemPrompt("reviewer", thread.task, config.projectContext),
    conversationContext: thread.toPromptContext(),
    apiKey: config.apiKey,
    model: config.agentModel,
  });
  thread.add({ role: "reviewer", content: reviewerResult.content, phase: "review" });

  const breakerResult = await callAgent({
    role: "breaker",
    systemPrompt: getSystemPrompt("breaker", thread.task, config.projectContext),
    conversationContext: thread.toPromptContext(),
    apiKey: config.apiKey,
    model: config.agentModel,
  });
  thread.add({ role: "breaker", content: breakerResult.content, phase: "review" });

  const reviewerApproved = reviewerResult.content.toLowerCase().includes("lgtm");
  const breakerPassed = breakerResult.content.toLowerCase().includes("passed");

  return reviewerApproved && breakerPassed;
}
```

```typescript
// src/orchestrator/run.ts
import { Thread } from "../conversation/thread.js";
import { renderConversation } from "../conversation/renderer.js";
import { renderSummary } from "../conversation/summary.js";
import { runDesignPhase, runBuildPhase, runReviewPhase } from "./phases.js";
import { writeCodeChanges } from "../execution/code-writer.js";
import type { GapsResult, ConversationMessage } from "../types.js";

export interface RunGapsInput {
  task: string;
  apiKey: string;
  projectDir: string;
  projectContext: string;
  architectModel?: string;
  agentModel?: string;
  maxDesignRounds?: number;
  maxReviewRounds?: number;
}

export interface RunGapsOutput extends GapsResult {
  conversationMd: string;
  summaryMd: string;
}

export async function runGaps(input: RunGapsInput): Promise<RunGapsOutput> {
  const thread = new Thread(input.task);
  const branchName = `gaps/${slugify(input.task)}-${formatDate(new Date())}`;

  const phaseConfig = {
    apiKey: input.apiKey,
    architectModel: input.architectModel ?? "claude-sonnet-4-6-20250514",
    agentModel: input.agentModel ?? "claude-sonnet-4-6-20250514",
    maxRounds: input.maxDesignRounds ?? 4,
    projectContext: input.projectContext,
  };

  // Phase 1: Design
  await runDesignPhase(thread, phaseConfig);

  // Phase 2: Build
  const codeChanges = await runBuildPhase(thread, phaseConfig);
  if (codeChanges.length > 0) {
    writeCodeChanges(input.projectDir, codeChanges);
  }

  // Phase 3: Review (with retry loop)
  const maxReviewRounds = input.maxReviewRounds ?? 3;
  for (let round = 0; round < maxReviewRounds; round++) {
    const approved = await runReviewPhase(thread, phaseConfig);
    if (approved) break;
    if (round < maxReviewRounds - 1) {
      const fixChanges = await runBuildPhase(thread, phaseConfig);
      if (fixChanges.length > 0) {
        writeCodeChanges(input.projectDir, fixChanges);
      }
    }
  }

  // Phase 4: Finalize
  const conversationMd = renderConversation(thread);
  const summaryMd = renderSummary(thread, branchName);

  return {
    task: input.task,
    messages: thread.messages,
    branch: branchName,
    startedAt: thread.startedAt,
    finishedAt: new Date(),
    stats: thread.getStats(),
    conversationMd,
    summaryMd,
  };
}

function slugify(text: string): string {
  return text.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/^-|-$/g, "").slice(0, 40);
}

function formatDate(date: Date): string {
  return date.toISOString().slice(0, 10);
}
```
